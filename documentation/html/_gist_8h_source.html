231  void performFFT()
232  {
233  // copy samples from audio frame
234  for (int i = 0;i < frameSize;i++)
235  {
236  fftIn[i][0] = (double) audioFrame[i];
237  fftIn[i][1] = (double) 0.0;
238  }
239 
240  // perform the FFT
241  fftw_execute(p);
242 
243  // store real and imaginary parts of FFT
244  for (int i = 0;i < frameSize;i++)
245  {
246  fftReal[i] = (T) fftOut[i][0];
247  fftImag[i] = (T) fftOut[i][1];
248  }
249 
250  // calculate the magnitude spectrum
251  for (int i = 0;i < frameSize/2;i++)
252  {
253  magnitudeSpectrum[i] = sqrt(pow(fftReal[i],2) + pow(fftImag[i],2));
254  }
255 
256  }
